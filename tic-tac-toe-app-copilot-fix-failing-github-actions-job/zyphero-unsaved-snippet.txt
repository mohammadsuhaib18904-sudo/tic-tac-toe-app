import 'package:flutter/material.dart';
import 'package:neon_tic_tac_toe/game_screen.dart';
import 'package:neon_tic_tac_toe/rock_paper_scissors_screen.dart';
import 'package:neon_tic_tac_toe/memory_match_screen.dart';
import 'package:neon_tic_tac_toe/settings_provider.dart';
import 'package:provider/provider.dart';

void main() {
  runApp(
    ChangeNotifierProvider(
      create: (context) => SettingsProvider(),
      child: const MyApp(),
    ),
  );
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Neon Game Center',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: const GameCenterScreen(),
      debugShowCheckedModeBanner: false,
    );
  }
}

class GameCenterScreen extends StatelessWidget {
  const GameCenterScreen({super.key});

  @override
  Widget build(BuildContext context) {
    return DefaultTabController(
      length: 3,
      child: Scaffold(
        backgroundColor: Colors.black,
        appBar: AppBar(
          backgroundColor: Colors.black,
          title: Text(
            'NEON GAME CENTER',
            style: TextStyle(
              color: Colors.cyan,
              fontWeight: FontWeight.bold,
              fontSize: 20,
              shadows: [
                Shadow(
                  color: Colors.cyan.withValues(alpha: 0.8),
                  blurRadius: 10,
                  offset: const Offset(0, 0),
                ),
              ],
            ),
          ),
          centerTitle: true,
          bottom: const TabBar(
            indicatorColor: Colors.cyan,
            labelColor: Colors.cyan,
            unselectedLabelColor: Colors.white70,
            tabs: [
              Tab(text: 'Tic Tac Toe'),
              Tab(text: 'Rock Paper Scissors'),
              Tab(text: 'Memory Match'),
            ],
          ),
        ),
        body: const TabBarView(
          children: [
            GameScreen(),
            RockPaperScissorsScreen(),
            MemoryMatchScreen(),
          ],
        ),
      ),
    );
  }
}
